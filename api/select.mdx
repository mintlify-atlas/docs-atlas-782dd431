# Select

A dropdown select component with support for single and multiple selection, built on Base UI's Select primitive.

## Base UI Primitive

Built on [`@base-ui/react/select`](https://base-ui.com/react/components/select)

## Import

```tsx
import { Select } from "@soft-ui/react/select"
```

## Usage

```tsx
import { Select } from "@soft-ui/react/select"

export default function Example() {
  return (
    <Select.Root defaultValue="apple">
      <Select.Trigger>
        <Select.Value placeholder="Select a fruit" />
        <Select.Icon />
      </Select.Trigger>
      <Select.Portal>
        <Select.Positioner>
          <Select.Popup>
            <Select.List>
              <Select.Item value="apple">
                <Select.ItemText>Apple</Select.ItemText>
                <Select.ItemIndicator />
              </Select.Item>
              <Select.Item value="banana">
                <Select.ItemText>Banana</Select.ItemText>
                <Select.ItemIndicator />
              </Select.Item>
            </Select.List>
          </Select.Popup>
        </Select.Positioner>
      </Select.Portal>
    </Select.Root>
  )
}
```

## Components

### Select.Root

<ParamField path="size" type="'s' | 'm' | 'l'" default="'m'">
  Control size.
  - `s`: 32px height
  - `m`: 36px height
  - `l`: 40px height
</ParamField>

<ParamField path="variant" type="'secondary' | 'tertiary'" default="'secondary'">
  Visual style variant.
  - `secondary`: Solid background
  - `tertiary`: Glass-morphic with backdrop blur
</ParamField>

<ParamField path="value" type="Value">
  Controlled value.
</ParamField>

<ParamField path="defaultValue" type="Value">
  Uncontrolled default value.
</ParamField>

<ParamField path="onValueChange" type="(value: Value) => void">
  Callback fired when value changes.
</ParamField>

<ParamField path="multiple" type="boolean">
  Enable multiple selection.
</ParamField>

<ParamField path="disabled" type="boolean">
  Disables the select.
</ParamField>

<ParamField path="name" type="string">
  Name attribute for form submission.
</ParamField>

### Select.Trigger

<ParamField path="className" type="string">
  Additional CSS classes.
</ParamField>

<ParamField path="unsafeClassName" type="string">
  Escape hatch for structural overrides.
</ParamField>

### Select.Value

<ParamField path="placeholder" type="React.ReactNode">
  Placeholder shown when no value is selected.
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes.
</ParamField>

### Select.Icon

<ParamField path="children" type="React.ReactNode">
  Custom icon. Defaults to expand/collapse icon.
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes.
</ParamField>

### Select.Positioner

<ParamField path="sideOffset" type="number" default="4">
  Distance from trigger.
</ParamField>

<ParamField path="collisionPadding" type="number" default="8">
  Padding from viewport edge.
</ParamField>

<ParamField path="alignItemWithTrigger" type="boolean" default="false">
  Align selected item with trigger.
</ParamField>

### Select.Item

<ParamField path="value" type="string">
  Value of the item.
</ParamField>

<ParamField path="disabled" type="boolean">
  Disables the item.
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes.
</ParamField>

### Select.Group / Select.GroupLabel

For grouping items with labels.

### Grouped Items
