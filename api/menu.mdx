# Menu

A dropdown menu component built on Base UI Menu primitive with support for items, groups, radio groups, checkboxes, and nested submenus.

## Compound Components

### Menu.Root

The root container that manages menu state.

<ParamField path="open" type="boolean">
  Controlled open state. When provided, the menu becomes controlled.
</ParamField>

<ParamField path="defaultOpen" type="boolean" default="false">
  Uncontrolled default open state.
</ParamField>

<ParamField path="onOpenChange" type="(open: boolean) => void">
  Callback fired when the open state changes.
</ParamField>

### Menu.Trigger

The trigger element that opens the menu.

<ParamField path="className" type="string">
  Additional CSS classes to apply.
</ParamField>

<ParamField path="asChild" type="boolean" default="false">
  Merge props onto the child element instead of rendering a button.
</ParamField>

### Menu.Portal

Portals the menu content to the end of the document body.

<ParamField path="keepMounted" type="boolean" default="false">
  Whether to keep the menu mounted in the DOM when closed.
</ParamField>

<ParamField path="container" type="HTMLElement | null">
  The container element to portal into. Defaults to `document.body`.
</ParamField>

### Menu.Positioner

Positions the menu relative to the trigger.

<ParamField path="side" type="'top' | 'right' | 'bottom' | 'left'" default="'bottom'">
  The side of the trigger to position the menu.
</ParamField>

<ParamField path="align" type="'start' | 'center' | 'end'" default="'start'">
  The alignment of the menu relative to the trigger.
</ParamField>

<ParamField path="sideOffset" type="number" default="4">
  The distance in pixels from the trigger.
</ParamField>

<ParamField path="alignOffset" type="number" default="0">
  The offset in pixels along the alignment axis.
</ParamField>

<ParamField path="collisionPadding" type="number | { top?: number; right?: number; bottom?: number; left?: number }" default="8">
  The padding in pixels from the viewport edges.
</ParamField>

<ParamField path="sticky" type="boolean" default="true">
  Whether the menu should stick to the trigger on scroll.
</ParamField>

### Menu.Popup

The menu popup container with list styles.

<ParamField path="className" type="string">
  Additional CSS classes to apply. Default: Surface overlay background with shadow.
</ParamField>

**Layout**: 
- Min-width: 220px
- Max-height: 320px (LIST_MAX_HEIGHT)
- Vertical scrolling when content overflows

### Menu.Arrow

Optional arrow pointing to the trigger.

<ParamField path="className" type="string">
  Additional CSS classes to apply. Default: `"fill-surface-overlay"`
</ParamField>

### MenuItem

A clickable menu item.

<ParamField path="onSelect" type="(event: Event) => void">
  Callback fired when the item is selected.
</ParamField>

<ParamField path="disabled" type="boolean" default="false">
  Whether the item is disabled.
</ParamField>

<ParamField path="closeOnSelect" type="boolean" default="true">
  Whether to close the menu when this item is selected.
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes to apply.
</ParamField>

### MenuPrefix

Leading content for a menu item (icon, avatar, etc.).

<ParamField path="className" type="string">
  Additional CSS classes to apply.
</ParamField>

### MenuSuffix

Trailing content for a menu item (shortcut, badge, chevron, etc.).

<ParamField path="className" type="string">
  Additional CSS classes to apply.
</ParamField>

### MenuSeparator

Visual separator between menu sections.

<ParamField path="className" type="string">
  Additional CSS classes to apply.
</ParamField>

### MenuGroup

Groups related menu items.

<ParamField path="className" type="string">
  Additional CSS classes to apply.
</ParamField>

### MenuGroupLabel

Label for a menu group.

<ParamField path="className" type="string">
  Additional CSS classes to apply. Default: Small subtle text with padding.
</ParamField>

### MenuEmpty

Empty state message when menu has no items.

<ParamField path="className" type="string">
  Additional CSS classes to apply.
</ParamField>

### Menu.RadioGroup

Groups radio items with single selection.

<ParamField path="value" type="string">
  The currently selected value.
</ParamField>

<ParamField path="onValueChange" type="(value: string) => void">
  Callback fired when the selection changes.
</ParamField>

### Menu.RadioItem

A radio menu item with indicator.

<ParamField path="value" type="string">
  The value of this radio item.
</ParamField>

<ParamField path="disabled" type="boolean" default="false">
  Whether the item is disabled.
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes to apply.
</ParamField>

**Indicator**: Shows checked circle icon (RiCheckboxCircleFill) when selected.

### Menu.CheckboxItem

A checkbox menu item with indicator.

<ParamField path="checked" type="boolean | 'indeterminate'">
  The checked state.
</ParamField>

<ParamField path="onCheckedChange" type="(checked: boolean | 'indeterminate') => void">
  Callback fired when the checked state changes.
</ParamField>

<ParamField path="disabled" type="boolean" default="false">
  Whether the item is disabled.
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes to apply.
</ParamField>

**Indicator**: Shows checked circle icon (RiCheckboxCircleFill) when checked.

### Menu.SubmenuRoot

Root container for a submenu.

<ParamField path="open" type="boolean">
  Controlled open state.
</ParamField>

<ParamField path="defaultOpen" type="boolean" default="false">
  Uncontrolled default open state.
</ParamField>

<ParamField path="onOpenChange" type="(open: boolean) => void">
  Callback fired when the submenu open state changes.
</ParamField>

### Menu.SubmenuTrigger

Trigger for opening a submenu.

<ParamField path="className" type="string">
  Additional CSS classes to apply.
</ParamField>

## Usage

```tsx
import { Menu, MenuItem, MenuPrefix, MenuSuffix, MenuSeparator } from '@soft-ui/react/menu'
import { Button } from '@soft-ui/react/button'
import { RiSettings4Line, RiUser3Line, RiLogoutBoxRLine } from '@soft-ui/icons'

function Example() {
  return (
    <Menu.Root>
      <Menu.Trigger asChild>
        <Button>Open Menu</Button>
      </Menu.Trigger>
      <Menu.Portal>
        <Menu.Positioner>
          <Menu.Popup>
            <MenuItem>
              <MenuPrefix><RiUser3Line /></MenuPrefix>
              Profile
              <MenuSuffix>⌘P</MenuSuffix>
            </MenuItem>
            <MenuItem>
              <MenuPrefix><RiSettings4Line /></MenuPrefix>
              Settings
            </MenuItem>
            <MenuSeparator />
            <MenuItem>
              <MenuPrefix><RiLogoutBoxRLine /></MenuPrefix>
              Logout
            </MenuItem>
          </Menu.Popup>
        </Menu.Positioner>
      </Menu.Portal>
    </Menu.Root>
  )
}
```

## Radio Group

```tsx
<Menu.Popup>
  <Menu.RadioGroup value={theme} onValueChange={setTheme}>
    <Menu.RadioItem value="light">Light</Menu.RadioItem>
    <Menu.RadioItem value="dark">Dark</Menu.RadioItem>
    <Menu.RadioItem value="system">System</Menu.RadioItem>
  </Menu.RadioGroup>
</Menu.Popup>
```

## Submenu

```tsx
<Menu.Popup>
  <MenuItem>Account</MenuItem>
  <Menu.SubmenuRoot>
    <Menu.SubmenuTrigger>
      Settings
      <MenuSuffix><RiArrowRightSLine /></MenuSuffix>
    </Menu.SubmenuTrigger>
    <Menu.Portal>
      <Menu.Positioner>
        <Menu.Popup>
          <MenuItem>General</MenuItem>
          <MenuItem>Privacy</MenuItem>
          <MenuItem>Notifications</MenuItem>
        </Menu.Popup>
      </Menu.Positioner>
    </Menu.Portal>
  </Menu.SubmenuRoot>
</Menu.Popup>
```
